    quantpylib.wrappers.bybit

    quantpylib.wrappers.bybit module is our official Bybit wrapper SDK implementing the endpoints for perpetuals trading. The library supports a fully asynchronous endpoint for efficiency and lightweight concurrency. The websocket manager handles reconnections and resubsriptions under network errors and upgrades.

    We will demonstrate usage of the library. On top of the endpoints exposed by the exchange, we have added a variety of utility functions - such as merging data from paginated responses.

    Examples

    We would demonstrate some endpoints. Refer to full documentation for details. 

    import os 
import pytz
import asyncio
from pprint import pprint
from datetime import datetime
from dotenv import load_dotenv
load_dotenv()

from quantpylib.standards import Period
from quantpylib.wrappers.bybit import Bybit

async def print_handler(msg): 
    print(msg)

async def main():
    '''using the SDK'''
    bybit = Bybit(
        key=os.getenv("BYBIT_KEY"),
        secret=os.getenv("BYBIT_SECRET"),
    )
    '''quantpylib.gateway endpoints'''
    await bybit.init_client()
    print(bybit.get_price_precision("BTCUSDT"))
    print(bybit.get_lot_precision("BTCUSDT"))

    '''ACCOUNT ENDPOINTS'''
    pprint(await bybit.account_balance())
    await bybit.account_fill_subscribe(handler=print_handler)

    '''EXCHANGE ENDPOINTS'''
    pprint(await bybit.contract_specifications())

    '''EXECUTOR ENDPOINTS'''
    cloid = bybit.rand_cloid()
    print(cloid)
    pprint(await bybit.get_all_mids())
    pprint(await bybit.limit_order(ticker="SOLUSDT",amount=1,price=99.99,cloid=cloid))
    pprint(await bybit.cancel_open_orders()) #or use ticker="SOLUSDT"
    pprint(await bybit.cancel_order(ticker="SOLUSDT",cloid=cloid)) #or use oid
    pprint(await bybit.market_order(ticker="SOLUSDT",amount=-0.5))
    pprint(await bybit.l2_book_get(ticker="BTCUSDT"))
    await bybit.l2_book_mirror(ticker="BTCUSDT",on_update=print_handler)
    await bybit.l2_book_subscribe(ticker="BTCUSDT",handler=print_handler)
    await bybit.trades_subscribe(ticker="BTCUSDT",handler=print_handler)

    '''ORDERS ENDPOINTS'''
    pprint(await bybit.order_query(oid='1234'))
    pprint(await bybit.orders_get())
    await bybit.orders_mirror(on_update=print_handler)
    await bybit.order_updates_subscribe(handler=print_handler)

    '''POSITIONS ENDPOINTS'''
    pprint(await bybit.positions_get())
    await bybit.positions_mirror(on_update=print_handler)

    '''DATAPOLLER ENDPOINTS'''
    pprint(await bybit.get_trade_bars(
        ticker='BTCUSDT',
        start=datetime(2021,1,1, tzinfo=pytz.utc),
        end=datetime.now(pytz.utc),
        granularity=Period.HOURLY,
        granularity_multiplier=1,
    ))

    '''raw exchange endpoints API - refer to docs for the other method implementations'''
    pprint(await bybit.instruments_info())
    await asyncio.sleep(1e9)

if __name__ == "__main__":
    asyncio.run(main())
